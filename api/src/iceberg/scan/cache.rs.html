28/// Manages the caching of [`BoundPredicate`] objects
29/// for [`PartitionSpec`]s based on partition spec id.
30#[derive(Debug)]
31pub(crate) struct PartitionFilterCache(RwLock<HashMap<i32, Arc<BoundPredicate>>>);
32
33impl PartitionFilterCache {
34    /// Creates a new [`PartitionFilterCache`]
35    /// with an empty internal HashMap.
36    pub(crate) fn new() -> Self {
37        Self(RwLock::new(HashMap::new()))
38    }
39
40    /// Retrieves a [`BoundPredicate`] from the cache
41    /// or computes it if not present.
42    pub(crate) fn get(
43        &self,
44        spec_id: i32,
45        table_metadata: &TableMetadataRef,
46        schema: &Schema,
47        case_sensitive: bool,
48        filter: BoundPredicate,
49    ) -> Result<Arc<BoundPredicate>> {
50        // we need a block here to ensure that the `read()` gets dropped before we hit the `write()`
51        // below, otherwise we hit deadlock
52        {
53            let read = self.0.read().map_err(|_| {
54                Error::new(
55                    ErrorKind::Unexpected,
56                    "PartitionFilterCache RwLock was poisoned",
57                )
58            })?;
59
60            if read.contains_key(&spec_id) {
61                return Ok(read.get(&spec_id).unwrap().clone());
62            }
63        }
64
65        let partition_spec = table_metadata
66            .partition_spec_by_id(spec_id)
67            .ok_or(Error::new(
68                ErrorKind::Unexpected,
69                format!("Could not find partition spec for id {spec_id}"),
70            ))?;
71
72        // A historical spec may reference a source column that was later dropped from the
73        // schema, which is a legitimate v2+ state. Such a spec cannot be resolved to a
74        // partition type, so it falls back to an always-true filter: files under the spec
75        // are not partition-pruned but still receive the row filter. Any other resolution
76        // failure is unexpected and propagates. The fallback is cached by spec id like any
77        // other filter; this is safe only because the cache lives per-scan in `PlanContext`
78        // with a fixed schema and predicate. Hoisting it to table or catalog scope would
79        // pin a spec to always-true even for a later scan whose schema could resolve it.
80        // TODO(https://github.com/apache/iceberg-rust/issues/2844): derive partition types from
81        // transforms where possible to restore pruning on a historical spec's still-live fields,
82        // which also makes the fallback per-term (dropped fields only) instead of per-spec.
83        let has_dropped_source_column = partition_spec
84            .fields()
85            .iter()
86            .any(|field| schema.field_by_id(field.source_id).is_none());
87
88        let partition_filter = if has_dropped_source_column {
89            BoundPredicate::AlwaysTrue
90        } else {
91            let partition_type = partition_spec.partition_type(schema)?;
92            let partition_fields = partition_type.fields().to_owned();
93            let partition_schema = Arc::new(
94                Schema::builder()
95                    .with_schema_id(partition_spec.spec_id())
96                    .with_fields(partition_fields)
97                    .build()?,
98            );
99
100            let mut inclusive_projection = InclusiveProjection::new(partition_spec.clone());
101
102            inclusive_projection
103                .project(&filter)?
104                .rewrite_not()
105                .bind(partition_schema.clone(), case_sensitive)?
106        };
107
108        let mut write = self.0.write().map_err(|_| {
109            Error::new(
110                ErrorKind::Unexpected,
111                "PartitionFilterCache RwLock was poisoned",
112            )
113        })?;
114
115        // Another thread may have populated the entry while the filter was computed without a
116        // lock held. Only warn and insert when this call is the one that populates it, so the
117        // always-true fallback is logged once per spec rather than once per racing thread.
118        let partition_filter = write.entry(spec_id).or_insert_with(|| {
119            if has_dropped_source_column {
120                tracing::warn!(
121                    spec_id,
122                    "Partition spec references a source column not in the scan schema; \
123                     skipping partition pruning for its manifests"
124                );
125            }
126
127            Arc::new(partition_filter)
128        });
129
130        Ok(partition_filter.clone())
131    }
132}

134/// Manages the caching of [`ManifestEvaluator`] objects
135/// for [`PartitionSpec`]s based on partition spec id.
136#[derive(Debug)]
137pub(crate) struct ManifestEvaluatorCache(RwLock<HashMap<i32, Arc<ManifestEvaluator>>>);
138
139impl ManifestEvaluatorCache {
140    /// Creates a new [`ManifestEvaluatorCache`]
141    /// with an empty internal HashMap.
142    pub(crate) fn new() -> Self {
143        Self(RwLock::new(HashMap::new()))
144    }
145
146    /// Retrieves a [`ManifestEvaluator`] from the cache
147    /// or computes it if not present.
148    pub(crate) fn get(
149        &self,
150        spec_id: i32,
151        partition_filter: Arc<BoundPredicate>,
152    ) -> Arc<ManifestEvaluator> {
153        // we need a block here to ensure that the `read()` gets dropped before we hit the `write()`
154        // below, otherwise we hit deadlock
155        {
156            let read = self
157                .0
158                .read()
159                .map_err(|_| {
160                    Error::new(
161                        ErrorKind::Unexpected,
162                        "ManifestEvaluatorCache RwLock was poisoned",
163                    )
164                })
165                .unwrap();
166
167            if read.contains_key(&spec_id) {
168                return read.get(&spec_id).unwrap().clone();
169            }
170        }
171
172        self.0
173            .write()
174            .map_err(|_| {
175                Error::new(
176                    ErrorKind::Unexpected,
177                    "ManifestEvaluatorCache RwLock was poisoned",
178                )
179            })
180            .unwrap()
181            .insert(
182                spec_id,
183                Arc::new(ManifestEvaluator::builder(partition_filter.as_ref().clone()).build()),
184            );
185
186        let read = self
187            .0
188            .read()
189            .map_err(|_| {
190                Error::new(
191                    ErrorKind::Unexpected,
192                    "ManifestEvaluatorCache RwLock was poisoned",
193                )
194            })
195            .unwrap();
196
197        read.get(&spec_id).unwrap().clone()
198    }
199}

264#[cfg(test)]
265mod tests {
266    use std::collections::HashMap;
267    use std::sync::Arc;
268
269    use super::PartitionFilterCache;
270    use crate::expr::{Bind, BoundPredicate, Reference};
271    use crate::spec::{
272        Datum, FormatVersion, NestedField, PrimitiveType, Schema, SortOrder, TableMetadataBuilder,
273        Transform, Type, UnboundPartitionSpec,
274    };
275
276    /// A historical spec whose source column was dropped from the current schema resolves to an
277    /// always-true filter, and the fallback is cached under the spec id.
278    #[test]
279    fn dropped_partition_source_column_falls_back_to_always_true() {
280        let schema = Schema::builder()
281            .with_fields(vec![
282                NestedField::required(1, "id", Type::Primitive(PrimitiveType::Long)).into(),
283                NestedField::required(2, "part", Type::Primitive(PrimitiveType::Long)).into(),
284            ])
285            .build()
286            .unwrap();
287
288        // Spec 0 partitions on `part` (id 2).
289        let spec = UnboundPartitionSpec::builder()
290            .with_spec_id(0)
291            .add_partition_field(2, "part", Transform::Identity)
292            .unwrap()
293            .build();
294
295        // Evolve the schema so that `part` is no longer present, leaving spec 0 unresolvable.
296        let evolved_schema = Schema::builder()
297            .with_fields(vec![
298                NestedField::required(1, "id", Type::Primitive(PrimitiveType::Long)).into(),
299            ])
300            .build()
301            .unwrap();
302        // Make an unpartitioned spec the default before dropping `part`, so spec 0 survives
303        // only as a historical spec that can no longer be resolved against the schema.
304        let table_metadata = Arc::new(
305            TableMetadataBuilder::new(
306                schema,
307                spec,
308                SortOrder::unsorted_order(),
309                "s3://bucket/test".to_string(),
310                FormatVersion::V2,
311                HashMap::new(),
312            )
313            .unwrap()
314            .add_default_partition_spec(UnboundPartitionSpec::builder().build())
315            .unwrap()
316            .add_current_schema(evolved_schema.clone())
317            .unwrap()
318            .build()
319            .unwrap()
320            .metadata,
321        );
322
323        let filter = Reference::new("id")
324            .greater_than_or_equal_to(Datum::long(5))
325            .bind(Arc::new(evolved_schema.clone()), true)
326            .unwrap();
327
328        let cache = PartitionFilterCache::new();
329        let partition_filter = cache
330            .get(0, &table_metadata, &evolved_schema, true, filter.clone())
331            .unwrap();
332        assert!(matches!(*partition_filter, BoundPredicate::AlwaysTrue));
333
334        // The fallback is cached, so a second lookup returns the same instance.
335        let cached = cache
336            .get(0, &table_metadata, &evolved_schema, true, filter)
337            .unwrap();
338        assert!(Arc::ptr_eq(&partition_filter, &cached));
339    }
